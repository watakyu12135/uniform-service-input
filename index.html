function updateInseamAreaForProduct(row) { if (!row || !row.dataset.productCode) return; const productCode = row.dataset.productCode; const productRows = getRowsByProductCode(productCode); if (productRows.length === 0) return; const currentInput = getProductInseamInput(productCode); const currentInseamCm = currentInput ? currentInput.value.trim() : ""; // 同じ商品の股下加工欄はいったん全部消す productRows.forEach(r => { const cell = r.querySelector(".cell-inseam"); if (!cell) return; cell.innerHTML = ""; }); // 股下加工欄は、同じ商品の最後の行に表示する const lastRow = productRows[productRows.length - 1]; const lastCell = lastRow.querySelector(".cell-inseam"); if (!lastCell) return; lastCell.innerHTML = buildInseamAreaHtml( lastRow.dataset.productCategory, lastRow.dataset.inseamAlterable || 0 ); const newInput = getProductInseamInput(productCode); if (newInput && currentInseamCm) { newInput.value = currentInseamCm; } updateInseamInputStateForProduct(productCode); }
